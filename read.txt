    <script src="https://code.jquery.com/jquery-3.7.1.min.js" integrity="sha256-/JqT3SQfawRcv/BIHPThkBvs0OEvtFFmqPF/lYI/Cxo=" crossorigin="anonymous"></script>
    <!-- Bootstrap JS for tooltips and utilities -->
    <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/js/bootstrap.bundle.min.js" integrity="sha384-YvpcrYf0tY3lHB60NNkmXc5s9fDVZLESaAA55NDzOxhy9GkcIdslK1eN7N6jIeHz" crossorigin="anonymous"></script>
    <!-- Select2 JS for searchable dropdowns -->
    <script src="https://cdn.jsdelivr.net/npm/select2@4.1.0-rc.0/dist/js/select2.min.js"></script>
    <!-- flatpickr JS for date picker -->
    <script src="https://cdn.jsdelivr.net/npm/flatpickr@4.6.13/dist/flatpickr.min.js"></script>
    <script>
        // Avoid jQuery conflicts
        const $ = jQuery.noConflict();

        $(document).ready(function () {
            console.log('DOM ready');

            try {
                // Calculate dynamic date range (today to past 30 days)
                const today = new Date();
                const past30Days = Array.from({ length: 30 }, (_, i) => {
                    const date = new Date(today);
                    date.setDate(today.getDate() - i);
                    return date.toISOString().split('T')[0];
                }).reverse();

                // Function to add status message with animation
                function addStatusMessage(text, delay) {
                    setTimeout(() => {
                        const $status = $('#status-messages');
                        const $message = $('<div class="status-message"></div>').text(text);
                        $status.append($message);
                        console.log('Status message added:', text); // Debug
                        $message.addClass('active');
                        setTimeout(() => {
                            $message.addClass('fade-out');
                            setTimeout(() => {
                                $message.remove();
                            }, 500);
                        }, 1500);
                    }, delay);
                }

                // Initialize Select2 for Rater Name (disabled initially)
                let $raterName = $('#rater-name');
                let $raterUpdate = $('#rater-update');
                if ($raterName.length) {
                    $raterName.select2({
                        placeholder: 'Select Rater',
                        allowClear: true,
                        width: '100%'
                    });
                    console.log('Rater Name Select2 initialized');
                    console.log('Rater update button present:', $raterUpdate.length > 0); // Debug
                    // Fetch raters from backend using AJAX
                    addStatusMessage('Loading rater names...', 0);
                    $.ajax({
                        url: '/get_raters',
                        method: 'GET',
                        success: function (data) {
                            console.log('Received raters:', data); // Debug response
                            $raterName.empty().append('<option></option>');
                            data.forEach(rater => {
                                $raterName.append(new Option(rater, rater));
                            });
                            // Enable Rater Name card after data is loaded
                            $raterName.prop('disabled', false);
                            $('#rater-card').removeClass('card-disabled').addClass('card-enabled');
                            addStatusMessage('Rater names loaded', 500);
                        },
                        error: function (xhr) {
                            console.error('Failed to fetch raters:', xhr);
                            alert('Failed to load raters. Please try again.');
                            addStatusMessage('Failed to load rater names', 500);
                        }
                    });
                } else {
                    console.error('Rater Name element not found');
                }

                // Initialize Select2 for Agent Name (disabled initially)
                let $agentName = $('#agent-name');
                let $agentUpdate = $('#agent-update');
                if ($agentName.length) {
                    $agentName.select2({
                        placeholder: 'Select Agents or All',
                        allowClear: true,
                        width: '100%',
                        data: [{ id: 'all', text: 'All', selected: true }]
                    });
                    console.log('Agent Name Select2 initialized');
                    console.log('Agent update button present:', $agentUpdate.length > 0); // Debug
                } else {
                    console.error('Agent Name element not found');
                }

                // Initialize flatpickr for Load Date (disabled initially)
                let $loadDate = $('#load-date');
                let datePicker = null;
                if ($loadDate.length) {
                    datePicker = flatpickr('#load-date', {
                        mode: 'multiple',
                        dateFormat: 'Y-m-d',
                        maxDate: today,
                        minDate: past30Days[0],
                        enable: [],
                        disable: past30Days
                    });
                    console.log('flatpickr initialized');
                } else {
                    console.error('Load Date element not found');
                }

                // Track selections for update buttons
                let selectedRater = null;
                let selectedAgents = [];

                // Enable update buttons based on selections
                $raterName.on('change', function () {
                    try {
                        selectedRater = $raterName.val();
                        console.log('Rater selected:', selectedRater); // Debug
                        // Enable update button if rater is selected
                        $raterUpdate.prop('disabled', !selectedRater);
                    } catch (e) {
                        console.error('Error in rater change handler:', e);
                    }
                });

                $agentName.on('change', function () {
                    try {
                        selectedAgents = $agentName.val() || [];
                        console.log('Agents selected:', selectedAgents); // Debug
                        // Enable update button if any agents are selected (including 'all')
                        $agentUpdate.prop('disabled', !selectedAgents.length);
                        // Remove 'All' option if specific agents are selected
                        if (selectedAgents.length && !selectedAgents.includes('all')) {
                            $agentName.find('option[value="all"]').remove();
                        }
                    } catch (e) {
                        console.error('Error in agent change handler:', e);
                    }
                });

                // Handle multi-line dates for >3 selections
                $loadDate.on('change', function () {
                    try {
                        if (!$loadDate.length) {
                            console.error('Load Date element not found in date change handler');
                            return;
                        }
                        const selectedDates = $loadDate.val() ? $loadDate.val().split(', ') : [];
                        console.log('Dates changed:', selectedDates); // Debug
                        $loadDate.toggleClass('multi-line', selectedDates.length > 3);
                    } catch (e) {
                        console.error('Error in date change handler:', e);
                    }
                });

                // Handle Rater Name update button click
                $raterUpdate.on('click', function () {
                    try {
                        // Hide tooltips
                        $('[data-bs-toggle="tooltip"]').tooltip('hide');
                        if (!selectedRater) {
                            alert('Please select a rater.');
                            return;
                        }
                        console.log('Rater update clicked for:', selectedRater); // Debug
                        addStatusMessage('Loading agent names...', 0);
                        // Clear and disable Agent Name and Load Date
                        $agentName.addClass('fade-out explode');
                        $loadDate.addClass('fade-out explode');
                        setTimeout(() => {
                            $agentName.val(null).trigger('change');
                            $agentName.select2({
                                data: [{ id: 'all', text: 'All', selected: true }],
                                width: '100%'
                            });
                            if (datePicker && typeof datePicker.clear === 'function') {
                                datePicker.clear();
                            } else {
                                $loadDate.val('');
                            }
                            $loadDate.removeClass('multi-line');
                            $agentName.prop('disabled', true);
                            $loadDate.prop('disabled', true);
                            $('#agent-card').removeClass('card-enabled').addClass('card-disabled');
                            $('#date-card').removeClass('card-enabled').addClass('card-disabled');
                            $agentUpdate.prop('disabled', true);
                            $agentName.removeClass('fade-out explode').addClass('fade-in');
                            $loadDate.removeClass('fade-out explode').addClass('fade-in');
                            setTimeout(() => {
                                $agentName.removeClass('fade-in');
                                $loadDate.removeClass('fade-in');
                            }, 500);
                            // Fetch agents from backend
                            $.ajax({
                                url: '/get_agents',
                                method: 'POST',
                                contentType: 'application/json',
                                data: JSON.stringify({ Rater_Name: selectedRater }),
                                success: function (data) {
                                    console.log('Received agents:', data); // Debug response
                                    $agentName.empty();
                                    data.forEach(agent => {
                                        $agentName.append(new Option(agent, agent, agent === 'all', agent === 'all'));
                                    });
                                    $agentName.trigger('change');
                                    // Enable Agent Name card
                                    $agentName.prop('disabled', false);
                                    $('#agent-card').removeClass('card-disabled').addClass('card-enabled');
                                    addStatusMessage('Agent names loaded', 500);
                                },
                                error: function (xhr) {
                                    console.error('Failed to fetch agents:', xhr);
                                    alert('Failed to load agents. Please try again.');
                                    addStatusMessage('Failed to load agent names', 500);
                                }
                            });
                        }, 500);
                    } catch (e) {
                        console.error('Error in rater update handler:', e);
                    }
                });

                // Handle Agent Name update button click
                $agentUpdate.on('click', function () {
                    try {
                        // Hide tooltips
                        $('[data-bs-toggle="tooltip"]').tooltip('hide');
                        if (!selectedAgents.length) {
                            alert('Please select specific agents or All.');
                            return;
                        }
                        console.log('Agent update clicked for:', selectedAgents); // Debug
                        addStatusMessage('Loading dates...', 0);
                        $loadDate.addClass('fade-out explode');
                        setTimeout(() => {
                            if (datePicker && typeof datePicker.clear === 'function') {
                                datePicker.clear();
                            } else {
                                $loadDate.val('');
                            }
                            $loadDate.removeClass('multi-line');
                            $loadDate.prop('disabled', true);
                            $('#date-card').removeClass('card-enabled').addClass('card-disabled');
                            $loadDate.removeClass('fade-out explode').addClass('fade-in');
                            setTimeout(() => {
                                $loadDate.removeClass('fade-in');
                            }, 500);
                            // Fetch dates for all selected agents
                            $.ajax({
                                url: '/get_dates',
                                method: 'POST',
                                contentType: 'application/json',
                                data: JSON.stringify({
                                    Rater_Name: selectedRater,
                                    Agent_Names: selectedAgents
                                }),
                                success: function (data) {
                                    console.log('Received dates:', data); // Debug response
                                    if (datePicker && typeof datePicker.set === 'function') {
                                        // Ensure dates are in YYYY-MM-DD format
                                        const validDates = data.filter(date => {
                                            const regex = /^\d{4}-\d{2}-\d{2}$/;
                                            return regex.test(date);
                                        });
                                        console.log('Valid dates for flatpickr:', validDates); // Debug
                                        datePicker.set('enable', validDates);
                                        datePicker.set('disable', past30Days.filter(d => !validDates.includes(d)));
                                    }
                                    // Enable Load Date card
                                    $loadDate.prop('disabled', false);
                                    $('#date-card').removeClass('card-disabled').addClass('card-enabled');
                                    addStatusMessage('Dates loaded', 500);
                                },
                                error: function (xhr) {
                                    console.error('Failed to fetch dates:', xhr);
                                    alert('Failed to load dates. Please try again.');
                                    addStatusMessage('Failed to load dates', 500);
                                }
                            });
                        }, 500);
                    } catch (e) {
                        console.error('Error in agent update handler:', e);
                    }
                });

                // Handle Clear button click
                $('#clear-btn').on('click', function () {
                    try {
                        console.log('Clear button clicked');
                        const $raterName = $('#rater-name');
                        const $loadDate = $('#load-date');
                        const $agentName = $('#agent-name');
                        const $overlay = $('#overlay');
                        const $raterUpdate = $('#rater-update');
                        const $agentUpdate = $('#agent-update');
                        if ($raterName.length && $loadDate.length && $agentName.length && $overlay.length) {
                            // Apply explode animation to all inputs
                            $raterName.addClass('fade-out explode');
                            $loadDate.addClass('fade-out explode');
                            $agentName.addClass('fade-out explode');
                            $overlay.addClass('hidden'); // Ensure overlay is hidden
                            addStatusMessage('Clearing selections...', 0);
                            setTimeout(() => {
                                // Reset inputs
                                $raterName.val('').trigger('change');
                                if (datePicker && typeof datePicker.clear === 'function') {
                                    console.log('Clearing datePicker');
                                    datePicker.clear();
                                } else {
                                    console.warn('datePicker.clear is not available');
                                    $loadDate.val('');
                                }
                                $loadDate.removeClass('multi-line');
                                $agentName.val(null).trigger('change');
                                $agentName.select2({
                                    data: [{ id: 'all', text: 'All', selected: true }],
                                    width: '100%'
                                });
                                // Disable Agent Name, Load Date, and update buttons
                                $agentName.prop('disabled', true);
                                $loadDate.prop('disabled', true);
                                $('#agent-card').removeClass('card-enabled').addClass('card-disabled');
                                $('#date-card').removeClass('card-disabled').addClass('card-disabled');
                                $raterUpdate.prop('disabled', true);
                                $agentUpdate.prop('disabled', true);
                                selectedRater = null;
                                selectedAgents = [];
                                // Reset animations
                                $raterName.removeClass('fade-out explode').addClass('fade-in');
                                $loadDate.removeClass('fade-out explode').addClass('fade-in');
                                $agentName.removeClass('fade-out explode').addClass('fade-in');
                                setTimeout(() => {
                                    $raterName.removeClass('fade-in');
                                    $loadDate.removeClass('fade-in');
                                    $agentName.removeClass('fade-in');
                                    addStatusMessage('Selections cleared', 500);
                                }, 500);
                            }, 500);
                        }
                    } catch (e) {
                        console.error('Error in clear button handler:', e);
                    }
                });

                // Handle Submit button click
                $('#submit-btn').on('click', function () {
                    try {
                        const $raterName = $('#rater-name');
                        const $loadDate = $('#load-date');
                        const $agentName = $('#agent-name');
                        const $overlay = $('#overlay');
                        const $progressBar = $('#progress-bar');
                        const $progressPercent = $('#progress-percent');
                        if ($raterName.length && $loadDate.length && $agentName.length && $overlay.length && $progressBar.length) {
                            const rater = $raterName.val();
                            if (!rater) {
                                alert('Please select a rater.');
                                return;
                            }
                            // Apply pulse animation to all inputs
                            $raterName.addClass('pulse');
                            $loadDate.addClass('pulse');
                            $agentName.addClass('pulse');
                            $overlay.removeClass('hidden');
                            $progressBar.css('width', '0');
                            addStatusMessage('Your input is submitted to the backend please wait...', 0);
                            addStatusMessage('Fetching your data please wait...', 2000);

                            // Poll progress from backend
                            let progressInterval = setInterval(function () {
                                $.get('/get_progress', function (data) {
                                    const progress = data.progress;
                                    $progressPercent.text(`${progress}%`);
                                    $progressBar.css('width', `${(progress / 100) * 300}px`);
                                    if (progress >= 100) {
                                        clearInterval(progressInterval);
                                    }
                                }).fail(function () {
                                    console.error('Failed to fetch progress');
                                });
                            }, 200);

                            // Submit data to fetch CSV using AJAX
                            $.ajax({
                                url: '/fetch_data',
                                method: 'POST',
                                contentType: 'application/json',
                                data: JSON.stringify({
                                    Rater_Name: rater,
                                    Agent_Names: JSON.stringify($agentName.val() || []),
                                    Load_Dates: JSON.stringify($loadDate.val() ? $loadDate.val().split(', ') : [])
                                }),
                                xhrFields: {
                                    responseType: 'blob'
                                },
                                success: function (data) {
                                    // Handle CSV download
                                    $raterName.removeClass('pulse').addClass('fade-out explode');
                                    $loadDate.removeClass('pulse').addClass('fade-out explode');
                                    $agentName.removeClass('pulse').addClass('fade-out explode');
                                    setTimeout(() => {
                                        const url = window.URL.createObjectURL(new Blob([data]));
                                        const a = document.createElement('a');
                                        a.href = url;
                                        a.download = 'data.csv';
                                        document.body.appendChild(a);
                                        a.click();
                                        document.body.removeChild(a);
                                        window.URL.revokeObjectURL(url);
                                        // Reset UI and hide overlay
                                        $raterName.removeClass('fade-out explode').addClass('fade-in');
                                        $loadDate.removeClass('fade-out explode').addClass('fade-in');
                                        $agentName.removeClass('fade-out explode').addClass('fade-in');
                                        $overlay.addClass('hidden');
                                        $progressBar.css('width', '0');
                                        setTimeout(() => {
                                            $raterName.removeClass('fade-in');
                                            $loadDate.removeClass('fade-in');
                                            $agentName.removeClass('fade-in');
                                            addStatusMessage('Data fetching completed', 500);
                                        }, 500);
                                    }, 500);
                                },
                                error: function (xhr) {
                                    console.error('Error fetching data:', xhr);
                                    alert('Error fetching data: ' + (xhr.responseJSON ? xhr.responseJSON.error : 'Unknown error'));
                                    $overlay.addClass('hidden');
                                    $progressBar.css('width', '0');
                                    $raterName.removeClass('pulse').addClass('fade-out explode');
                                    $loadDate.removeClass('pulse').addClass('fade-out explode');
                                    $agentName.removeClass('pulse').addClass('fade-out explode');
                                    setTimeout(() => {
                                        $raterName.removeClass('fade-out explode').addClass('fade-in');
                                        $loadDate.removeClass('fade-out explode').addClass('fade-in');
                                        $agentName.removeClass('fade-out explode').addClass('fade-in');
                                        setTimeout(() => {
                                            $raterName.removeClass('fade-in');
                                            $loadDate.removeClass('fade-in');
                                            $agentName.removeClass('fade-in');
                                            addStatusMessage('Failed to fetch data', 500);
                                        }, 500);
                                    }, 500);
                                    clearInterval(progressInterval);
                                }
                            });
                        }
                    } catch (e) {
                        console.error('Error in submit button handler:', e);
                    }
                });

                // Initialize Bootstrap tooltips
                try {
                    $('[data-bs-toggle="tooltip"]').tooltip();
                    console.log('Tooltips initialized');
                } catch (e) {
                    console.error('Error initializing tooltips:', e);
                }
            } catch (e) {
                console.error('Global script error:', e);
            }
        });
    </script>
</body>
