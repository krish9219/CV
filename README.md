find /search/path -type f -mtime +30 -exec sh -c 'for f do bytes=$(stat -c%s "$f"); days_old=$(( ( $(date +%s) - $(stat -c %Y "$f") ) / 86400 )); printf "%s | %.2f MB | %.2f GB | %s | %s | %s | %s days old\n" "$f" "$(echo "$bytes/1024/1024" | bc -l)" "$(echo "$bytes/1024/1024/1024" | bc -l)" "$(stat -c %w "$f" 2>/dev/null || echo N/A)" "$(stat -c %y "$f")" "$(stat -c %U "$f")" "$days_old"; done' sh {} + > /save/location/report.txt
